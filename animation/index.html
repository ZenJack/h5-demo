.animation-div:hover {
        /*
         * myfirst: 动画名称;
         * 2s：动画时长;
         * ease：前后缓慢。[linear, ease, ease-in, ease-out, ease-in-out]
         * infinite: 无限次数, 默认为 0;
         * alternate: 反向播放动画, 默认为 normal;
         */
        animation: myfirst 2s ease-in-out infinite alternate;
        transition: all 2s;
        transform-origin: bottom right;
      }
      @keyframes myfirst {
        from {
          left: 0;
        }
        to {
          left: 800px;
          background-color: blue;
          transform: rotate(720deg) scale(2, 2);
        }
      }
